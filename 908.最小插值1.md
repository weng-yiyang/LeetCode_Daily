代码：

```C++
class Solution {
public:
    int smallestRangeI(vector<int>& nums, int k) {
       int minnum=INT_MAX;
       int maxnum=0;
       for(auto const n:nums)
       {
           minnum=min(n,minnum);
           maxnum=max(n,maxnum);
       }
       if(maxnum-minnum-2*k>=0)
       {
           return maxnum-minnum-2*k;
       }
       return 0;
    }
};
```
